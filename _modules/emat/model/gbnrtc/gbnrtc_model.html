[docs]    def run_model(self, scenario, policy):
        """
        Runs an experiment through core model.

        This method overloads the `run_model` method given in
        the EMA Workbench, and provides the correct execution
        of the GBNRTC model within that framework.

        For each experiment, the core model is called to:

            1. set experiment variables
            2. run the experiment
            3. run post-processors associated with specified
               performance measures
            4. (optionally) archive model outputs
            5. record performance measures to database

        Note that this method does *not* return any outcomes.
        Outcomes are instead written into self.outcomes_output,
        and can be retrieved from there.

        Args:
            scenario (Scenario): A dict-like object that
                has key-value pairs for each uncertainty.
            policy (Policy): A dict-like object that
                has key-value pairs for each lever.

        Raises:
            UserWarning: If there are no experiments associated with
                this type.

        """

        design = None
        archive = True

        _logger.debug("run_core_model read_experiment_parameters")

        experiment_id = self.db.read_experiment_id(self.scope.name, None, scenario, policy)

        if experiment_id is not None and self.allow_short_circuit:
            # opportunity to short-circuit run by loading pre-computed values.
            precomputed = self.db.read_experiment_measures(
                self.scope.name,
                design=None,
                experiment_id=experiment_id,
            )
            if not precomputed.empty:
                self.outcomes_output = dict(precomputed.iloc[0])
                return

        if experiment_id is None:
            experiment_id = self.db.write_experiment_parameters_1(
                self.scope.name, 'ad hoc', scenario, policy
            )

        xl = {}
        xl.update(scenario)
        xl.update(policy)

        m_names = [m.name for m in self.scope._m_list]

        m_out = pd.DataFrame()

        _logger.debug(f"run_core_model setup {experiment_id}")
        self.setup(xl)

        _logger.debug(f"run_core_model run {experiment_id}")
        self.run()

        _logger.debug(f"run_core_model post_process {experiment_id}")
        self.post_process(xl, m_names)

        _logger.debug(f"run_core_model wrap up {experiment_id}")
        m_di = self.import_meas(m_names)
        m_df = pd.DataFrame(m_di, index=[experiment_id])

        self.outcomes_output = m_di

        _logger.debug(f"run_core_model write db {experiment_id}")
        self.db.write_experiment_measures(self.scope.name, self.metamodel_id, m_df)
        m_out = pd.concat([m_out, m_df])

        if archive:
            _logger.debug(f"run_core_model archive {experiment_id}")
            self.archive(xl, self.get_ex_archive_path(self.scope.name, experiment_id))

[docs]    def start_transcad(self):
        """
        Launch TransCAD and call methods to reset scenario
        """
        # wack transCAD if it is open (cannot restart model)
        print("[(re)starting TransCAD instances]")
        os.system("TASKKILL /F /IM tcw.exe")
        time.sleep(10)
        os.startfile(self.tc_ver_path + 'tcw.exe')
        time.sleep(20)

        self.tc = cp.Gisdk(
            "TransCAD",
            log_file=self.mod_path + "TC_log.txt",
            search_path=self.tc_ver_path)
        
        if self.tc is None:
            print("ERROR: failed to attach to a TransCAD instance")
            sys.exit()

    

[docs]    def setup(self, xl_di: dict):
        """
        Configure the core model with the experiment variable values

        Args:
            xl_di (dict): experiment variables including both exogenous
                uncertainty and policy levers

        Raises:
            KeyError: if experiment variable defined is not supported
                by the core model
        """

        self.start_transcad()
        # initilize scenario output space
        ret = self.tc.RunMacro("Restore Defaults", 
                               self.exp_param_ui, 
                               self.mod_path_tc)

        if ret != 0:
            raise SystemError("Error in restore defaults {0}".format(ret))

        variables_done = ['constant']
        for xl in xl_di:
            # skip if variable handled as part of a combined setting
            if xl in variables_done: continue
        
            print("\t\t\tSetting experiment variable for {0} to {1}".format(
                xl, xl_di[xl]))
            
            try:
                func, macro = self.__METHOD_MACRO_BY_EVAR[xl]
                variables_done += func(self, macro, xl, xl_di)
            except KeyError:
                print("Experiment variable method not available")
                raise

            raise UserWarning("ERROR: failed to attach to a TransCAD instance")

        self.tc.RunMacro("G30 File Close All", self.tdm_ui)
        self.tc.RunMacro("GBNRTC Model - AutoRun 2050", self.tdm_ui)   

        

[docs]    def post_process(self,
                     xl_di: dict, 
                     m_names: List[str],
                     output_path=None):
        """
        Runs post processors associated with measures.

        For the GBNRTC model, this method calls
        TransCAD macros to generate output files.

        The model should have previously been executed using
        the `run` method.

        Args:
            xl_di (dict):
                Dictionary of experiment variables - indices
                are variable names, values are the experiment settings
            m_names (List[str]):
                List of measures to be processed
            output_path (str):
                Path to model outputs - if set to none
                will use local values

        Raises:
            KeyError:
                If post process is not available for specified
                measure
        """

        print("Running post process scripts at {0}"
              .format(time.strftime("%Y_%m_%d %H%M%S")))
        
        if self.tc is None:
            self.start_transcad()
        
        if output_path is None:
            output_path = self.mod_path
            
        output_path = output_path.replace("\\", "\\\\") + "\\\\"
        
        pm_done = []
        for pm in m_names:
            # skip if performance measure handled by other macro
            if pm in pm_done: continue
            
            try:
                func, macro = self.__TRANSCAD_MACRO_BY_PM[pm]
                pm_done += func(self, macro, xl_di, output_path)
            except KeyError:
                print("Post process method for pm {0} not available"
                      .format(pm))
                raise

    

[docs]    @copydoc(AbstractCoreModel.import_meas)
    def import_meas(self, 
                    m_names: List[str],
                    output_path=None):
        print("Importing performance measures at {0}"
              .format(time.strftime("%Y_%m_%d %H%M%S")))  
 
        if output_path is None:
            output_path = os.path.join(self.mod_path, r"outputs")
        else:
            output_path = os.path.join(output_path, r"outputs")
        
        pm_done = {}
        for pm in m_names:
            # skip if performance measure import already complete
            if pm in pm_done: continue
        
            try:
                func, file = self.__IMPORT_METHOD_BY_PM[pm]
            except KeyError:
                print("Import method for pm {0} not available"
                      .format(pm))  
                raise
                
            # check that file exists
            full_file_path = os.path.join(output_path, file)
            if not os.path.exists(full_file_path):
                raise KeyError("PM file {0} not found".format(full_file_path))
            
            pm_done.update(func(self, file, output_path, m_names))

        return pm_done

    def __set_simple_evar(self, macro, evar, exp_var):       
        ''' call TransCAD macro and return variable to completed list'''
        self.tc.RunMacro("G30 File Close All", self.exp_param_ui)  
        ret = self.tc.RunMacro(macro, 
                               self.exp_param_ui, 
                               self.mod_path_tc,
                               float(exp_var[evar]))
        if ret != 0:
            raise SystemError("Error {2} in setting {0} to {1}" 
                              .format(evar, exp_var[evar], ret)) 
        return [evar]
        
    
    def __set_kenslrt_strategy(self, macro, evar, exp_var): 
        ''' call TransCAD macro and return both strategies to completed list'''
        # kensington and lrt strategy are interdependent
        kens = int(exp_var['Kensington Decommissioning'])
        lrt = int(exp_var['LRT Extension'])

        self.tc.RunMacro("G30 File Close All", self.exp_param_ui)  
        ret = self.tc.RunMacro(macro, 
                               self.exp_param_ui, 
                               self.mod_path_tc,
                               kens, lrt)
        
        ret = self.tc.RunMacro("VA", 
                               self.exp_param_ui, 
                               self.mod_path_tc,
                               float(exp_var['Shared Mobility']))
        if ret != 0:
            raise SystemError("Error in setting LRT / Kensington to {0}/{1}" 
                              .format(lrt, kens)) 
            
        return ['Kensington Decommissioning', 'LRT Extension']

        return self.__PM_BY_TRANSCAD_MACRO[macro][1]

    def __import_single_row(self, file, out_path, requested_perf_meas_ids):
        '''Import for csv's with single row of values that includes header'''

        pm_ret = {}
        
        imp = pd.read_csv(os.path.join(out_path, file))
        
        file_pm = self.__IMPORT_PM_BY_FILE[file][1]
        
        for pm in list(set(requested_perf_meas_ids).intersection(file_pm)):
            pm_ret[pm] = imp[self.__SINGLE_ROW_COLUMN_BY_PM[pm]].sum()

        return pm_ret
  
    def __import_row_column(self, file, out_path, requested_perf_meas_ids):
        '''
        Import for csv's with multiple row/columns

        Includes row header and column index'''

        pm_ret = {}
        
        imp = pd.read_csv(os.path.join(out_path, file), index_col=0)
        
        file_pm = self.__IMPORT_PM_BY_FILE[file][1]
        
        for pm in list(set(requested_perf_meas_ids).intersection(file_pm)):
            row, col = self.__MULT_ROW_COLUMN_BY_PM[pm]
            pm_ret[pm] = imp.loc[row, col].sum()

        return pm_ret    
  
    def __import_district_od(self, file, out_path, requested_perf_meas_ids):
        '''custom import method for OD flow data'''

        pm_ret = {}
        
        od = pd.read_csv(os.path.join(out_path, file), 
                        names=range(1, 11))
        
        od_pm = self.__IMPORT_PM_BY_FILE[file][1]
        
        # need to chagne mapping statement here if more ODs are added to 
        # PM list
        od_d = {}
        for d in range(1, 11):
            od_d[od_pm[d-1]] = od.loc[1, d]
        
        for pm in list(set(requested_perf_meas_ids).intersection(od_pm)):
            pm_ret[pm] = od_d[pm].sum()

        return pm_ret

    def __import_vmtvht(self, file, out_path, requested_perf_meas_ids):
        '''a little different process because we need to process the
        raw results form the model'''

        pm_ret = {}
        
        vmtvht= pd.read_csv(os.path.join(out_path, file), 
                            names=['Daily Auto VMT', 'Daily Truck VMT', 
                                     'Daily Total VMT', 'Daily Auto VHT', 
                                     'Daily Truck VHT', 'Daily Total VHT', 
                                    'AM Auto VMT', 'AM Truck VMT', 'AM Total VMT', 
                                    'MD Auto VMT', 'MD Truck VMT', 'MD Total VMT', 
                                    'PM Auto VMT', 'PM Truck VMT', 'PM Total VMT', 
                                    'NT Auto VMT', 'NT Truck VMT', 'NT Total VMT', 
                                    'AM Auto VHT', 'AM Truck VHT', 'AM Total VHT', 
                                    'MD Auto VHT', 'MD Truck VHT', 'MD Total VHT', 
                                    'PM Auto VHT', 'PM Truck VHT', 'PM Total VHT', 
                                    'NT Auto VHT', 'NT Truck VHT', 'NT Total VHT', 
                                    'Daily Total Volume', 'Daily Total Count', 
                                    'Daily Auto Volume', 'Daily Auto Count', 
                                    'Daily Truck Volume', 'Daily Truck Count'])
                        
        if "Region-wide VMT" in requested_perf_meas_ids:
            pm_ret["Region-wide VMT"] = vmtvht.sum(axis=0)['Daily Total VMT']

        if "Total Auto VMT" in requested_perf_meas_ids:
            pm_ret["Total Auto VMT"] = vmtvht.sum(axis=0)['Daily Auto VMT']
            
        if "Total Truck VMT" in requested_perf_meas_ids:
            pm_ret["Total Truck VMT "] = vmtvht.sum(axis=0)['Daily Truck VMT']            

        if "Interstate + Expressway + Ramp/Connector VMT" in requested_perf_meas_ids:
            pm_ret["Interstate + Expressway + Ramp/Connector VMT"] = (
                    vmtvht['Daily Total VMT'][1:2].sum() + 
                    vmtvht['Daily Total VMT'][6:8].sum() + 
                    vmtvht['Daily Total VMT'][21:22].sum())
            
        if "Major and Minor Arterials VMT" in requested_perf_meas_ids:
            pm_ret["Major and Minor Arterials VMT"] = (
                    vmtvht['Daily Total VMT'][3:4].sum() + 
                    vmtvht['Daily Total VMT'][23].sum())         
  
        return pm_ret
